#define _CRT_SECURE_NO_WARNINGS   
#include <stdio.h>     
#include <stdlib.h>

void Img(char *name);



int main(void)
{
	Img((char *)"pipi.txt");
}

void Img(char* name)
{
	{
		char ch, * name_tmp = (char *)malloc(sizeof(char) * 10);
		name_tmp = name;
		FILE* fp = fopen(name, "rt");
		if (fp == NULL) {
			printf("파일 오픈 실패 !\n");
		}
		while (1) {
			ch = fgetc(fp);
			if (ch == EOF)
				break;
			putchar(ch);
		}
		fclose(fp);
	}
}
